with aircraft_seats_count as 
(
SELECT count(s.seat_no) count_seats,a.aircraft_code
FROM dst_project.seats s 
join dst_project.aircrafts a on s.aircraft_code=a.aircraft_code
group by a.aircraft_code
),
mainds as (
SELECT *
FROM dst_project.flights
WHERE departure_airport = 'AAQ'
  AND (date_trunc('month', scheduled_departure) in ('2017-01-01','2017-02-01', '2017-12-01'))
  AND status not in ('Cancelled')
limit 20
)
select mainds.flight_id,count(tf.ticket_no) count_seats_sold,aircraft_seats_count.count_seats count_seats_total
from dst_project.ticket_flights tf
join mainds on tf.flight_id=mainds.flight_id
join aircraft_seats_count on aircraft_seats_count.aircraft_code=mainds.aircraft_code
group by mainds.flight_id,aircraft_seats_count.count_seats
